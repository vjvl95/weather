### 1. 공통 애니메이션 유틸

```typescript
// File: src/shared/lib/animations.ts
import { Easing, type EasingFunction } from 'react-native-reanimated';

/**
 * 이동/전환에 사용할 easing 배열
 * 매번 같은 느낌이 반복되지 않도록 시드 기반 랜덤 선택
 */
export const EASING_VARIANTS: EasingFunction[] = [
  Easing.inOut(Easing.ease),
  Easing.inOut(Easing.quad),
  Easing.bezier(0.25, 0.1, 0.25, 1),
  Easing.bezier(0.42, 0, 0.58, 1),
  Easing.inOut(Easing.cubic),
];

/**
 * 시드 기반 easing 선택
 */
export function pickEasing(seed: number): EasingFunction {
  const index = Math.abs(seed) % EASING_VARIANTS.length;
  return EASING_VARIANTS[index];
}
```

### 2. Reduced Motion 전면 대응

**통일 기준 (모든 WO에서 이 규칙을 따름):**

| 규칙 | 설명 |
|------|------|
| `withRepeat(-1)` | **금지** (CLAUDE.md ANR 규칙) |
| idle 완전 비활성화 | **금지** (캐릭터가 죽어 보임) |
| JS 타이머 느린 1회 시퀀스 | **채택** — `setTimeout`/`setInterval`로 느린 1회 시퀀스를 재트리거 |
| 탭/쓰다듬기 반응 | **항상 유지** (1회성, ANR 위험 없음) |
| blink | **유지** (이미 JS 타이머 기반) |
| 이펙트 파티클 | 수 감소 + 속도 50% → JS 타이머 반복 |
| 자동 이동 | **비활성화** (center-hero 고정) |

> WO-008에서 `if (reduceMotion) return;`으로 건너뛴 idle은
> 이 WO에서 JS 타이머 패턴으로 교체합니다.

V1 아키텍처 섹션 8-3의 규칙을 각 컴포넌트에 적용:

#### CharacterSprite.tsx 수정
```typescript
// Reduced Motion이 켜져 있을 때:
// - ⚠️ withRepeat(-1) 사용 금지 (CLAUDE.md ANR 규칙 준수)
// - 무한 반복 대신 1회성 느린 시퀀스로 대체
// - idleRotate 비활성화
// - tapBounce는 유지 (사용자 직접 트리거이므로, 1회성)
// - petTilt 최대 각도 clamp ±2deg

if (reduceMotion) {
  // withRepeat(-1) 금지 → 1회 느린 시퀀스로 "살아있는 느낌"만 유지
  translateY.value = withSequence(
    withTiming(-params.floatAmplitude * 0.15, {
      duration: params.floatDuration * 2,
      easing: Easing.inOut(Easing.ease),
    }),
    withTiming(0, {
      duration: params.floatDuration * 2,
      easing: Easing.inOut(Easing.ease),
    }),
  );
  // idleRotate는 0으로 유지
  // 반복이 필요하면 JS 타이머로 재트리거 (UI 스레드 withRepeat 회피)
}
```

### 3. 성능 최적화

```typescript
// 각 컴포넌트에 적용할 최적화:
```

### 4. 진입/복귀 타이밍

```typescript
// 앱 진입 시 시퀀스:
//
// 1. 배경 먼저 표시 (즉시)
// 2. 300ms 후 캐릭터 fade-in + 아래에서 위로 슬라이드
// 3. 500ms 후 idle 모션 시작
// 4. 1000ms 후 첫 말풍선 표시
//
// 구현: CharacterView에 진입 시퀀스 상태 관리

// CharacterView.tsx에 추가:
// showBubble은 useCharacterStore에서 직접 가져옴 (useCharacterState 반환값에 없음)
const { showBubble } = useCharacterStore();
const [phase, setPhase] = useState<'bg' | 'character' | 'idle' | 'ready'>('bg');

useEffect(() => {
  const timers = [
    setTimeout(() => setPhase('character'), 300),
    setTimeout(() => setPhase('idle'), 800),
    setTimeout(() => {
      setPhase('ready');
      showBubble();
    }, 1300),
  ];
  return () => timers.forEach(clearTimeout);
}, [showBubble]);

// phase별 렌더링:
// 'bg': 배경만
// 'character': 배경 + 캐릭터 (fade-in)
// 'idle': 배경 + 캐릭터 (모션 시작)
// 'ready': 전체 기능 활성화
```
