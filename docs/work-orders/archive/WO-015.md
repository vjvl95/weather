### 1. useCharacterDirector — 자동 이동/말풍선 제어 훅

```typescript
// File: src/features/character/lib/useCharacterDirector.ts
import { useEffect, useRef, useCallback } from 'react';
import { useCharacterStore } from '../model/useCharacterStore';
import { useReducedMotion } from 'react-native-reanimated';
import {
  ANCHOR_POINTS,
  AUTO_MOVE_CONFIG,
  BUBBLE_CONFIG,
} from '@shared/config';
import type { AnchorId } from '@shared/types';

/** 시드 기반 RNG (CLAUDE.md: Math.random 금지) */
function mulberry32(seed: number): () => number {
  return () => {
    let t = (seed += 0x6d2b79f5);
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}

const rng = mulberry32(Date.now());

/**
 * 앵커 간 인접 그래프 (허용 경로만 정의)
 *
 * ⚠️ 자유 선택이 아닌 인접 앵커만 이동 가능
 * 이유: 긴 대각선 점프나 UI를 가로지르는 이동 방지
 * V1 아키텍처 "짧고 통제된 이동" 원칙 준수
 *
 * 시각적 그래프:
 *
 *   [top-left] ──── [center] ──── [top-right]
 *       │              │               │
 *   [left-peek] ──── [center] ──── [right-peek]
 *                      │
 *                 [bottom-rest]
 */
const ANCHOR_ADJACENCY: Record<AnchorId, AnchorId[]> = {
  'center-hero':     ['top-left-drift', 'top-right-drift', 'left-peek', 'right-peek', 'bottom-rest'],
  'top-left-drift':  ['center-hero', 'left-peek'],
  'top-right-drift': ['center-hero', 'right-peek'],
  'left-peek':       ['center-hero', 'top-left-drift', 'bottom-rest'],
  'right-peek':      ['center-hero', 'top-right-drift', 'bottom-rest'],
  'bottom-rest':     ['center-hero', 'left-peek', 'right-peek'],
};

/**
 * 캐릭터 자동 행동 디렉터
 *
 * 책임:
 * 1. 일정 간격으로 앵커 포인트 사이 자동 이동
 * 2. 방치 시 자동 말풍선 표시
 * 3. 야간 빈도 조절
 * 4. Reduced Motion 대응
 *
 * 이 훅은 CharacterView에서 한 번만 호출
 */
export function useCharacterDirector() {
  const reduceMotion = useReducedMotion();
  const {
    presentation,
    runtime,
    setAnchor,
    showBubble,
  } = useCharacterStore();

  const moveTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const idleTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const isNight = presentation?.weatherCondition === 'night';

  // --- 자동 이동 ---

  /** 다음 앵커 포인트 선택 (인접 앵커만 허용) */
  const pickNextAnchor = useCallback((): AnchorId => {
    const currentAnchor = runtime.anchorId;
    const neighbors = ANCHOR_ADJACENCY[currentAnchor];
    const index = Math.floor(rng() * neighbors.length);
    return neighbors[index];
  }, [runtime.anchorId]);

  /** 랜덤 이동 간격 계산 */
  const getNextInterval = useCallback((): number => {
    const { intervalMin, intervalMax, nightSlowdown } = AUTO_MOVE_CONFIG;
    const base = intervalMin + rng() * (intervalMax - intervalMin);
    return isNight ? base * nightSlowdown : base;
  }, [isNight]);

  /** 자동 이동 스케줄링 */
  const scheduleNextMove = useCallback(() => {
    if (reduceMotion) return; // Reduced Motion: 자동 이동 비활성화 (WO-016 통일 기준)

    moveTimerRef.current = setTimeout(() => {
      const nextAnchor = pickNextAnchor();
      setAnchor(nextAnchor);
      scheduleNextMove(); // 재귀적 스케줄링
    }, getNextInterval());
  }, [reduceMotion, pickNextAnchor, getNextInterval, setAnchor]);

  // --- 방치 시 자동 말풍선 ---

  const scheduleIdleCheckIn = useCallback(() => {
    idleTimerRef.current = setTimeout(() => {
      showBubble();
      scheduleIdleCheckIn(); // 재스케줄
    }, BUBBLE_CONFIG.idleCheckInDelay);
  }, [showBubble]);

  // --- 생명주기 ---

  useEffect(() => {
    scheduleNextMove();
    scheduleIdleCheckIn();

    return () => {
      if (moveTimerRef.current) clearTimeout(moveTimerRef.current);
      if (idleTimerRef.current) clearTimeout(idleTimerRef.current);
    };
  }, [scheduleNextMove, scheduleIdleCheckIn]);

  // 탭/상호작용 시 이동 타이머 리셋 + 중앙 복귀
  const onInteraction = useCallback(() => {
    // 기존 이동 타이머 리셋
    if (moveTimerRef.current) clearTimeout(moveTimerRef.current);
    // 중앙으로 복귀
    setAnchor('center-hero');
    // 새 이동 스케줄 시작
    scheduleNextMove();

    // idle 타이머도 리셋
    if (idleTimerRef.current) clearTimeout(idleTimerRef.current);
    scheduleIdleCheckIn();
  }, [setAnchor, scheduleNextMove, scheduleIdleCheckIn]);

  /** 현재 앵커의 화면 좌표 (비율) */
  const currentAnchorPosition = ANCHOR_POINTS[runtime.anchorId];

  return {
    currentAnchorPosition,
    onInteraction,
  };
}
```

### 2. 공통 애니메이션 유틸 (pickEasing)

```typescript
// File: src/shared/lib/animations.ts
import { Easing, type EasingFunction } from 'react-native-reanimated';

/**
 * 이동/전환에 사용할 easing 배열
 * 매번 같은 느낌이 반복되지 않도록 시드 기반 랜덤 선택
 */
export const EASING_VARIANTS: EasingFunction[] = [
  Easing.inOut(Easing.ease),
  Easing.inOut(Easing.quad),
  Easing.bezier(0.25, 0.1, 0.25, 1),
  Easing.bezier(0.42, 0, 0.58, 1),
  Easing.inOut(Easing.cubic),
];

/**
 * 시드 기반 easing 선택
 */
export function pickEasing(seed: number): EasingFunction {
  const index = Math.abs(seed) % EASING_VARIANTS.length;
  return EASING_VARIANTS[index];
}
```

### 3. CharacterView — 앵커 컨테이너 패턴 도입

```typescript
// File: src/widgets/character-view/ui/CharacterView.tsx (수정)
//
// ⚠️ 핵심 설계: 앵커 이동은 CharacterSprite가 아닌 "컨테이너"에 적용
//
// 이유:
// - 그림자(CharacterShadow)는 본체와 같은 앵커 위치를 공유해야 함
// - 그림자는 idleOffsetY만 보고 깊이감을 표현
// - 앵커 이동을 스프라이트에 넣으면 그림자가 중앙에 남거나 별개로 움직임
//
// 구조:
//   <WeatherBackground>
//     <AmbientEffects />
//     <AnchorContainer>      ← 앵커 이동 애니메이션이 여기에 적용
//       <SpeechBubble />
//       <CharacterSprite />  ← idle + 제스처만 담당 (위치 이동 없음)
//       <CharacterShadow />  ← idleOffsetY만 반영 (이전과 동일)
//     </AnchorContainer>
//   </WeatherBackground>

import { useCharacterDirector } from '@features/character';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withTiming,
} from 'react-native-reanimated';
import { useEffect, useRef } from 'react';
import { Dimensions } from 'react-native';
import { pickEasing } from '@shared/lib/animations';
import { AUTO_MOVE_CONFIG } from '@shared/config';

const { width: SCREEN_W, height: SCREEN_H } = Dimensions.get('window');

export function CharacterView() {
  const { currentAnchorPosition, onInteraction } = useCharacterDirector();
  // ... 기존 useCharacterState 유지

  // 앵커 이동용 SharedValue
  const anchorX = useSharedValue(currentAnchorPosition.x * SCREEN_W);
  const anchorY = useSharedValue(currentAnchorPosition.y * SCREEN_H);
  const moveCountRef = useRef(0);

  // 앵커 변경 감지 → 컨테이너 이동 애니메이션
  //
  // ⚠️ useAnimatedReaction은 SharedValue 감시용 — currentAnchorPosition은
  //     일반 JS 객체이므로 useEffect가 안정적
  useEffect(() => {
    const easing = pickEasing(moveCountRef.current);
    moveCountRef.current += 1;
    anchorX.value = withTiming(currentAnchorPosition.x * SCREEN_W, {
      duration: AUTO_MOVE_CONFIG.moveDuration,
      easing,
    });
    anchorY.value = withTiming(currentAnchorPosition.y * SCREEN_H, {
      duration: AUTO_MOVE_CONFIG.moveDuration,
      easing,
    });
  }, [currentAnchorPosition, anchorX, anchorY]);

  // 컨테이너 스타일 (앵커 이동)
  const containerStyle = useAnimatedStyle(() => ({
    position: 'absolute',
    transform: [
      { translateX: anchorX.value - SCREEN_W / 2 },
      { translateY: anchorY.value - SCREEN_H / 2 },
    ],
  }));

  return (
    <WeatherBackground colors={presentation.backgroundColors}>
      <AmbientEffects effectPreset={presentation.effectPreset} />

      {/* 앵커 컨테이너: 스프라이트 + 그림자가 함께 이동 */}
      <Animated.View style={containerStyle} className="items-center justify-center">
        <SpeechBubble ... />
        <CharacterSprite ... />    {/* idle + 제스처만 */}
        <CharacterShadow idleOffsetY={idleOffsetY} />  {/* 깊이감만 */}
      </Animated.View>
    </WeatherBackground>
  );
}
```

> **CharacterSprite는 수정 최소화**: anchorX/Y prop 추가 없음.
> 스프라이트는 여전히 idleFloat, tapBounce, petTilt만 담당.
> 위치 이동은 부모 컨테이너가 처리하므로 관심사가 깔끔하게 분리됨.
