### 2. SpeechBubble — 말풍선

```typescript
// File: src/widgets/character-view/ui/SpeechBubble.tsx
import { View, Text } from 'react-native';
import Animated, { FadeIn, FadeOut } from 'react-native-reanimated';
import { useEffect } from 'react';
import { BUBBLE_CONFIG } from '@shared/config';

interface SpeechBubbleProps {
  message: string | null;
  visible: boolean;
  onAutoHide: () => void;
}

/**
 * 캐릭터 위에 표시되는 말풍선
 *
 * - Spring fade-in, 200ms fade-out
 * - 자동 숨김 타이머 내장
 */
export function SpeechBubble({ message, visible, onAutoHide }: SpeechBubbleProps) {
  // 자동 숨김 타이머
  useEffect(() => {
    if (!visible) return;

    const timer = setTimeout(() => {
      onAutoHide();
    }, BUBBLE_CONFIG.autoHideDuration);

    return () => clearTimeout(timer);
  }, [visible, onAutoHide]);

  if (!visible || !message) return null;

  return (
    <Animated.View
      entering={FadeIn.springify().damping(15)}
      exiting={FadeOut.duration(200)}
      className="bg-white rounded-2xl px-5 py-3 mx-8 mb-4 shadow-md"
    >
      <Text className="text-base text-gray-800 text-center leading-6">
        {message}
      </Text>
      {/* 말풍선 꼬리 */}
      <View
        className="absolute -bottom-2 left-1/2 -ml-2 w-0 h-0"
        style={{
          borderLeftWidth: 8,
          borderRightWidth: 8,
          borderTopWidth: 8,
          borderLeftColor: 'transparent',
          borderRightColor: 'transparent',
          borderTopColor: 'white',
        }}
      />
    </Animated.View>
  );
}
```

### 3. CharacterSprite — 캐릭터 본체 + 모션

```typescript
// File: src/widgets/character-view/ui/CharacterSprite.tsx
import { Image, ImageSourcePropType } from 'react-native';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withRepeat,
  withSequence,
  withTiming,
  withSpring,
  useReducedMotion,
  Easing,
} from 'react-native-reanimated';
import { Gesture, GestureDetector } from 'react-native-gesture-handler';
import { useEffect } from 'react';
import type { MotionPreset } from '@shared/types';
import { MOTION_PARAMS } from '@shared/config';
import { mulberry32 } from '@shared/lib';

const AnimatedImage = Animated.createAnimatedComponent(Image);

interface CharacterSpriteProps {
  image: ImageSourcePropType;
  motionPreset: MotionPreset;
  onTap: () => void;
  onPetStart: () => void;
  onPetEnd: () => void;
}

/**
 * 몽글이 본체 렌더링 + 제스처 + 모션
 *
 * Phase 1 모션:
 * - idleFloat: 상하 부유 (motionPreset에 따라 진폭/속도 변화)
 * - idleRotate: 미세 회전
 * - tapBounce: 탭 시 통통 튀기
 * - petTilt: 쓰다듬기 시 기울어짐
 */
export function CharacterSprite({
  image,
  motionPreset,
  onTap,
  onPetStart,
  onPetEnd,
}: CharacterSpriteProps) {
  const reduceMotion = useReducedMotion();
  const params = MOTION_PARAMS[motionPreset];

  // 공유 애니메이션 값
  const translateY = useSharedValue(0);
  const scale = useSharedValue(1);
  const rotate = useSharedValue(0);
  const idleRotate = useSharedValue(0);
  const blinkEyeOpacity = useSharedValue(0); // 눈 감은 오버레이의 opacity (0=눈 뜸, 1=눈 감음)

  // idleFloat: 둥실둥실 떠다니기
  //
  // ⚠️ Reduced Motion — Phase 1 임시 구현
  //
  //   Phase 1(이 WO): idle을 건너뜀 (return)
  //   최종 기준(WO-016): JS 타이머 느린 1회 시퀀스로 교체
  //
  //   이 return은 WO-016에서 반드시 교체됩니다.
  //   최종 통일 기준은 WO-016의 "Reduced Motion 통일 기준" 표를 참조.
  //
  useEffect(() => {
    if (reduceMotion) return; // ← Phase 1 임시. WO-016에서 JS 타이머 패턴으로 교체 예정

    translateY.value = withRepeat(
      withSequence(
        withTiming(-params.floatAmplitude, {
          duration: params.floatDuration / 2,
          easing: Easing.inOut(Easing.ease),
        }),
        withTiming(0, {
          duration: params.floatDuration / 2,
          easing: Easing.inOut(Easing.ease),
        }),
      ),
      -1,
      true,
    );

    // idleRotate: 미세 좌우 흔들림
    idleRotate.value = withRepeat(
      withSequence(
        withTiming(params.rotateAmplitude, {
          duration: params.floatDuration,
          easing: Easing.inOut(Easing.ease),
        }),
        withTiming(-params.rotateAmplitude, {
          duration: params.floatDuration,
          easing: Easing.inOut(Easing.ease),
        }),
      ),
      -1,
      true,
    );
  }, [reduceMotion, params, translateY, idleRotate]);

  // blink: 불규칙 간격의 눈 깜빡임 (생명감 핵심 모션)
  // V1 아키텍처 섹션 8-1: "불규칙 간격의 표정 변화"
  //
  // ⚠️ 전체 캐릭터 opacity로 하면 몸 전체가 사라졌다 나타남 → 부자연스러움
  // 구현: 눈 위치에 캐릭터 배경색과 같은 타원형 오버레이를 겹침
  //   blinkEyeOpacity: 0 = 눈 뜸 (오버레이 투명), 1 = 눈 감음 (오버레이 불투명)
  //   오버레이는 캐릭터 이미지 위에 absolute 배치
  //
  // 간격: 2~6초 랜덤 (JS 타이머 — withRepeat 회피)
  useEffect(() => {
    const rng = mulberry32(Date.now());
    const doBlink = () => {
      blinkEyeOpacity.value = withSequence(
        withTiming(1, { duration: 75 }),  // 눈 감기
        withTiming(0, { duration: 75 }),  // 눈 뜨기
      );
      const nextDelay = 2000 + rng() * 4000;
      timer = setTimeout(doBlink, nextDelay);
    };
    let timer = setTimeout(doBlink, 2000 + rng() * 4000);
    return () => clearTimeout(timer);
  }, [blinkEyeOpacity]);

  // 탭 제스처: 통통 튀기
  const tapGesture = Gesture.Tap().onEnd(() => {
    scale.value = withSequence(
      withSpring(1.15, { damping: 4, stiffness: 300 }),
      withSpring(1, { damping: 8, stiffness: 200 }),
    );
    onTap();
  });

  // 팬 제스처: 쓰다듬기
  // minDistance: 10 → 미세한 손떨림으로 탭이 씹히는 것 방지
  // rotate clamp: ±15deg → 과도한 기울어짐 방지
  const MAX_PET_TILT_DEG = 15;
  const panGesture = Gesture.Pan()
    .minDistance(10)
    .onStart(() => {
      onPetStart();
    })
    .onUpdate((e) => {
      'worklet';
      const raw = e.translationX * 0.1;
      rotate.value = Math.max(-MAX_PET_TILT_DEG, Math.min(MAX_PET_TILT_DEG, raw));
    })
    .onEnd(() => {
      rotate.value = withSpring(0, { damping: 8 });
      onPetEnd();
    });

  // Exclusive: 팬이 minDistance 미만이면 탭으로 처리
  const composedGesture = Gesture.Exclusive(panGesture, tapGesture);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateY: translateY.value },
      { scale: scale.value },
      { rotate: `${rotate.value + idleRotate.value}deg` },
    ],
  }));

  // blink 오버레이 스타일 (눈 영역만 가림)
  const blinkOverlayStyle = useAnimatedStyle(() => ({
    opacity: blinkEyeOpacity.value,
  }));

  return (
    <GestureDetector gesture={composedGesture}>
      <Animated.View style={[{ width: 200, height: 200 }, animatedStyle]}>
        <AnimatedImage
          source={image}
          style={{ width: 200, height: 200 }}
          resizeMode="contain"
        />
        {/* blink 오버레이: 눈 위치에 캐릭터 배경색 타원을 겹침 */}
        {/* 위치/크기는 에셋별로 조정 필요 — 일러스트 눈 위치에 맞춤 */}
        <Animated.View
          style={[
            {
              position: 'absolute',
              top: 72,      // 눈 위치 (에셋별 조정)
              left: 62,
              width: 28,
              height: 10,
              borderRadius: 5,
              backgroundColor: '#F5E6D0', // 몽글이 피부색 (에셋별 조정)
            },
            blinkOverlayStyle,
          ]}
        />
        <Animated.View
          style={[
            {
              position: 'absolute',
              top: 72,
              right: 62,
              width: 28,
              height: 10,
              borderRadius: 5,
              backgroundColor: '#F5E6D0',
            },
            blinkOverlayStyle,
          ]}
        />
      </Animated.View>
    </GestureDetector>
  );
}
```

### 4. CharacterView — 레이어 조합

```typescript
// File: src/widgets/character-view/ui/CharacterView.tsx
import { View } from 'react-native';
import { useCharacterState } from '@features/character';
import { WeatherBackground } from './WeatherBackground';
import { CharacterSprite } from './CharacterSprite';
import { SpeechBubble } from './SpeechBubble';

/**
 * 캐릭터 전체 화면 위젯
 *
 * V1 레이어 구조 (Phase 1):
 * 1. WeatherBackgroundLayer ✅
 * 2. AmbientEffectLayer    → WO-014 (Phase 2)
 * 3. CharacterShadowLayer  → WO-014 (Phase 2)
 * 4. CharacterSpriteLayer  ✅
 * 5. DialogueLayer         ✅
 * 6. WeatherInfoOverlay    → WO-010 (홈 페이지)
 */
export function CharacterView() {
  const {
    presentation,
    runtime,
    onCharacterTap,
    onPetStart,
    onPetEnd,
    hideBubble,
  } = useCharacterState();

  if (!presentation) return null;

  return (
    <WeatherBackground colors={presentation.backgroundColors}>
      <View className="flex-1 items-center justify-center">
        {/* 말풍선 (캐릭터 위) */}
        <SpeechBubble
          message={runtime.currentBubbleMessage}
          visible={runtime.isBubbleVisible}
          onAutoHide={hideBubble}
        />

        {/* 몽글이 본체 (중앙 고정 — Phase 1) */}
        <CharacterSprite
          image={presentation.asset.image}
          motionPreset={presentation.motionPreset}
          onTap={onCharacterTap}
          onPetStart={onPetStart}
          onPetEnd={onPetEnd}
        />
      </View>
    </WeatherBackground>
  );
}
```
