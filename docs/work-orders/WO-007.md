## 작업 내용

### 1. Character 타입 정의

```typescript
// File: src/features/character/model/types.ts
import type { WeatherCondition } from '@shared/types';
import type { CharacterState } from '@shared/config';

export interface CharacterStoreState {
  /** 현재 날씨 상태 */
  currentCondition: WeatherCondition;
  /** 현재 캐릭터 상태 (이미지, 배경, 메시지 등) */
  characterState: CharacterState | null;
  /** 말풍선 표시 여부 */
  isBubbleVisible: boolean;
}

export interface CharacterStoreActions {
  setCondition: (condition: WeatherCondition) => void;
  setCharacterState: (state: CharacterState) => void;
  showBubble: () => void;
  hideBubble: () => void;
}
```

### 2. Character Zustand Store

```typescript
// File: src/features/character/model/useCharacterStore.ts
import { create } from 'zustand';
import type { CharacterStoreState, CharacterStoreActions } from './types';
import type { CharacterState } from '@shared/config';
import type { WeatherCondition } from '@shared/types';
import { WEATHER_CHARACTER_MAP } from '@shared/config';

type CharacterStore = CharacterStoreState & CharacterStoreActions;

export const useCharacterStore = create<CharacterStore>((set) => ({
  // State
  currentCondition: 'sunny',
  characterState: WEATHER_CHARACTER_MAP.sunny,
  isBubbleVisible: false,

  // Actions
  setCondition: (condition: WeatherCondition) =>
    set({
      currentCondition: condition,
      characterState: WEATHER_CHARACTER_MAP[condition],
    }),

  setCharacterState: (state: CharacterState) =>
    set({ characterState: state }),

  showBubble: () => set({ isBubbleVisible: true }),
  hideBubble: () => set({ isBubbleVisible: false }),
}));
```

### 3. 날씨 → 캐릭터 상태 연동 훅

```typescript
// File: src/features/character/lib/useCharacterState.ts
import { useEffect, useCallback } from 'react';
import { useCharacterStore } from '../model/useCharacterStore';
import { WEATHER_CHARACTER_MAP } from '@shared/config';
import type { WeatherCondition } from '@shared/types';

/**
 * 날씨 상태를 받아 캐릭터 상태를 업데이트하는 훅
 * pages/widgets에서 weather 데이터와 연결하여 사용
 */
export function useCharacterState() {
  const {
    currentCondition,
    characterState,
    isBubbleVisible,
    setCondition,
    showBubble,
    hideBubble,
  } = useCharacterStore();

  /** 날씨 변경 시 캐릭터 업데이트 */
  const updateForWeather = useCallback(
    (condition: WeatherCondition) => {
      if (condition !== currentCondition) {
        setCondition(condition);
        // 날씨 바뀌면 말풍선 자동 표시
        showBubble();
      }
    },
    [currentCondition, setCondition, showBubble],
  );

  /** 캐릭터 탭 시 말풍선 토글 */
  const onCharacterTap = useCallback(() => {
    if (isBubbleVisible) {
      hideBubble();
    } else {
      showBubble();
    }
  }, [isBubbleVisible, showBubble, hideBubble]);

  /** 현재 상태의 메시지 가져오기 */
  const getMessage = useCallback((): string => {
    return characterState?.message ?? '';
  }, [characterState]);

  return {
    currentCondition,
    characterState,
    isBubbleVisible,
    updateForWeather,
    onCharacterTap,
    getMessage,
  };
}
```
