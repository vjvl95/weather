### 1. 날씨 배경 컴포넌트

```typescript
// File: src/widgets/character-view/ui/WeatherBackground.tsx
import { View } from 'react-native';
import { LinearGradient } from 'expo-linear-gradient';
import Animated, { useAnimatedStyle, withTiming } from 'react-native-reanimated';

const AnimatedLinearGradient = Animated.createAnimatedComponent(LinearGradient);

interface WeatherBackgroundProps {
  backgroundColor: string;
  children: React.ReactNode;
}

/** 날씨에 따라 배경색이 부드럽게 전환되는 배경 */
export function WeatherBackground({ backgroundColor, children }: WeatherBackgroundProps) {
  // 그라데이션 상단/하단 색상 계산
  const topColor = backgroundColor;
  const bottomColor = adjustBrightness(backgroundColor, -20);

  return (
    <LinearGradient
      colors={[topColor, bottomColor]}
      className="flex-1"
    >
      {children}
    </LinearGradient>
  );
}

/** HEX 색상 밝기 조절 유틸 */
function adjustBrightness(hex: string, amount: number): string {
  const num = parseInt(hex.replace('#', ''), 16);
  const r = Math.min(255, Math.max(0, ((num >> 16) & 0xff) + amount));
  const g = Math.min(255, Math.max(0, ((num >> 8) & 0xff) + amount));
  const b = Math.min(255, Math.max(0, (num & 0xff) + amount));
  return `#${((r << 16) | (g << 8) | b).toString(16).padStart(6, '0')}`;
}
```

### 2. 말풍선 컴포넌트

```typescript
// File: src/widgets/character-view/ui/SpeechBubble.tsx
import { View, Text } from 'react-native';
import Animated, {
  useAnimatedStyle,
  withSpring,
  withDelay,
  FadeIn,
  FadeOut,
} from 'react-native-reanimated';

interface SpeechBubbleProps {
  message: string;
  visible: boolean;
}

export function SpeechBubble({ message, visible }: SpeechBubbleProps) {
  if (!visible) return null;

  return (
    <Animated.View
      entering={FadeIn.springify().damping(15)}
      exiting={FadeOut.duration(200)}
      className="bg-white rounded-2xl px-5 py-3 mx-8 mb-4 shadow-md"
    >
      <Text className="text-base text-gray-800 text-center leading-6">
        {message}
      </Text>
      {/* 말풍선 꼬리 삼각형 */}
      <View
        className="absolute -bottom-2 left-1/2 -ml-2 w-0 h-0"
        style={{
          borderLeftWidth: 8,
          borderRightWidth: 8,
          borderTopWidth: 8,
          borderLeftColor: 'transparent',
          borderRightColor: 'transparent',
          borderTopColor: 'white',
        }}
      />
    </Animated.View>
  );
}
```

### 3. 캐릭터 스프라이트 (애니메이션 포함)

```typescript
// File: src/widgets/character-view/ui/CharacterSprite.tsx
import { Image, ImageSourcePropType } from 'react-native';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withRepeat,
  withSequence,
  withTiming,
  withSpring,
  useReducedMotion,
  Easing,
} from 'react-native-reanimated';
import { Gesture, GestureDetector } from 'react-native-gesture-handler';
import { useEffect } from 'react';

const AnimatedImage = Animated.createAnimatedComponent(Image);

interface CharacterSpriteProps {
  image: ImageSourcePropType;
  onTap: () => void;
}

export function CharacterSprite({ image, onTap }: CharacterSpriteProps) {
  const reduceMotion = useReducedMotion();

  // 애니메이션 값
  const translateY = useSharedValue(0);
  const scale = useSharedValue(1);
  const rotate = useSharedValue(0);

  // 기본 애니메이션: 둥실둥실 떠다니기
  useEffect(() => {
    if (reduceMotion) return; // ANR 방지 (CLAUDE.md 규칙)

    translateY.value = withRepeat(
      withSequence(
        withTiming(-8, { duration: 1500, easing: Easing.inOut(Easing.ease) }),
        withTiming(0, { duration: 1500, easing: Easing.inOut(Easing.ease) }),
      ),
      -1, // 무한 반복
      true,
    );
  }, [reduceMotion, translateY]);

  // 탭 제스처: 통통 튀기
  const tapGesture = Gesture.Tap().onEnd(() => {
    scale.value = withSequence(
      withSpring(1.15, { damping: 4, stiffness: 300 }),
      withSpring(1, { damping: 8, stiffness: 200 }),
    );
    onTap();
  });

  // 팬 제스처: 쓰다듬기 (좌우 흔들림)
  const panGesture = Gesture.Pan()
    .onUpdate((e) => {
      rotate.value = e.translationX * 0.1;
    })
    .onEnd(() => {
      rotate.value = withSpring(0, { damping: 8 });
    });

  const composedGesture = Gesture.Race(tapGesture, panGesture);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateY: translateY.value },
      { scale: scale.value },
      { rotate: `${rotate.value}deg` },
    ],
  }));

  return (
    <GestureDetector gesture={composedGesture}>
      <AnimatedImage
        source={image}
        style={[{ width: 200, height: 200 }, animatedStyle]}
        resizeMode="contain"
      />
    </GestureDetector>
  );
}
```

### 4. Character View 조합 위젯

```typescript
// File: src/widgets/character-view/ui/CharacterView.tsx
import { View } from 'react-native';
import { useCharacterState } from '@features/character';
import { WeatherBackground } from './WeatherBackground';
import { CharacterSprite } from './CharacterSprite';
import { SpeechBubble } from './SpeechBubble';

export function CharacterView() {
  const {
    characterState,
    isBubbleVisible,
    onCharacterTap,
    getMessage,
  } = useCharacterState();

  if (!characterState) return null;

  return (
    <WeatherBackground backgroundColor={characterState.backgroundColor}>
      <View className="flex-1 items-center justify-center">
        {/* 말풍선 */}
        <SpeechBubble
          message={getMessage()}
          visible={isBubbleVisible}
        />

        {/* 몽글이 캐릭터 */}
        <CharacterSprite
          image={characterState.image}
          onTap={onCharacterTap}
        />
      </View>
    </WeatherBackground>
  );
}
```
