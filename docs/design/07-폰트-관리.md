## 폰트 로딩 패턴

```typescript
// app/_layout.tsx
import * as SplashScreen from 'expo-splash-screen';
import { useFonts } from 'expo-font';

SplashScreen.preventAutoHideAsync();

export default function RootLayout() {
  const [fontsLoaded, fontsError] = useFonts({
    'Pretendard-Bold': require('../assets/fonts/Pretendard-Bold.ttf'),
    'Pretendard-Regular': require('../assets/fonts/Pretendard-Regular.ttf'),
  });

  useEffect(() => {
    if (fontsLoaded || fontsError) {
      SplashScreen.hideAsync();
    }
  }, [fontsLoaded, fontsError]);

  if (!fontsLoaded && !fontsError) return null;

  return <Stack />;
}
```

## Skia 폰트 FOUT 방지

Skia 폰트와 RN 폰트는 **별도로** 로딩됩니다.

```typescript
// Skia 폰트도 완료까지 대기해야 FOUT(Flash of Unstyled Text) 방지 가능
const [isSkiaFontLoaded, setIsSkiaFontLoaded] = useState(false);

// SplashScreen은 fontsLoaded AND isSkiaFontLoaded 둘 다 완료 후 숨김
useEffect(() => {
  if (fontsLoaded && isSkiaFontLoaded) {
    SplashScreen.hideAsync();
  }
}, [fontsLoaded, isSkiaFontLoaded]);
```
